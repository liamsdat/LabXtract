import pydicom
from labxtract.core.extractor import LabXtractEngine

def integrate_with_dicom(lab_file, dicom_file):
    """–ò–Ω—Ç–µ–≥—Ä–∞—Ü–∏—è –ª–∞–±–æ—Ä–∞—Ç–æ—Ä–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö —Å DICOM"""

    reports = engine.process_file(lab_file)
    
    # –ß–∏—Ç–∞–µ–º DICOM —Ñ–∞–π–ª
    ds = pydicom.dcmread(dicom_file)
    
    # –î–æ–±–∞–≤–ª—è–µ–º –ª–∞–±–æ—Ä–∞—Ç–æ—Ä–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ –≤ DICOM

        if report.patient.full_name:
            ds.PatientName = report.patient.full_name
        
        # –î–æ–±–∞–≤–ª—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –∞–Ω–∞–ª–∏–∑–æ–≤ –∫–∞–∫ –ø—Ä–∏–≤–∞—Ç–Ω—ã–µ —Ç–µ–≥–∏

            value = f"{test.name}: {test.numeric_value} {test.unit}"
            ds[tag] = pydicom.dataelem.DataElement(tag, 'LO', value)
    
    return ds
